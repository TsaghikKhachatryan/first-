function bubble_sort(arr) {
  var len = arr.length - 1;
  for (var i = 0; i < len; ++i) {
    for (var j = 0; j < len - i; ++j) {
      if (arr[j] > arr[j + 1]) {
        var temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
  }
}

function n_smallest(n, arr) {
  bubble_sort(arr);
  return arr[n - 1];
}
